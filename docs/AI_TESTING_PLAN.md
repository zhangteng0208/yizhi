## 7. 测试工具

### 7.1 自动化测试脚本

```bash
#!/bin/bash
# test-ai-depth.sh

API_URL="http://localhost:3000/api/divination/ai-stream"
DEPTHS=("simple" "normal" "pro")

for depth in "${DEPTHS[@]}"; do
  echo "Testing depth: $depth"
  for i in {1..10}; do
    echo "  Test $i..."
    time curl -X POST $API_URL \
      -H "Content-Type: application/json" \
      -d "{\"type\":\"bazi\",\"data\":{...},\"extraParams\":{\"depth\":\"$depth\"}}" \
      > "result_${depth}_${i}.json"
  done
done
```

### 7.2 结果分析脚本

```javascript
// analyze-results.js
const fs = require('fs');

const depths = ['simple', 'normal', 'pro'];
const results = {};

depths.forEach(depth => {
  results[depth] = {
    count: 0,
    totalChars: 0,
    totalTokens: 0,
    parseErrors: 0,
  };

  for (let i = 1; i <= 10; i++) {
    const file = `result_${depth}_${i}.json`;
    try {
      const data = JSON.parse(fs.readFileSync(file, 'utf8'));
      results[depth].count++;
      results[depth].totalChars += JSON.stringify(data).length;
      // 统计 Token（需要后端返回）
    } catch (e) {
      results[depth].parseErrors++;
    }
  }
});

console.log('Test Results:', results);
```
